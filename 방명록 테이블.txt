-- 시퀀스 생성
create sequence seq_visit_idx;

-- 테이블 생성
create table VISIT (
	idx NUMBER(3) primary key,
	name VARCHAR2(50), -- 작성자 이름
	content VARCHAR2(1000), -- 내용
	pwd VARCHAR2(50), -- 비밀번호
	ip VARCHAR2(20), -- ip
	regidate DATE -- 등록날짜	
);

-- 샘플데이터
insert into VISIT values (seq_visit_idx.nextVal, '홍길동', '테스트 내용입니다.', '1111', '127.0.0.1', sysdate);
insert into VISIT values (seq_visit_idx.nextVal, '이길동', 'Hello Spring', '1111', '127.0.0.1', sysdate);
